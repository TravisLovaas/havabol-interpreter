def String whatSign(Int x):

	if x == 0:
		return "zero";
	endif;

	if x > 0:
		return "positive";
	else:
		return "negative";
	endif;

enddef;

Int x = 0;
Int y = 5;
Int z = -5;

print("x = " # x # " -> " # whatSign(x));
print("y = " # y # " -> " # whatSign(y));
print("z = " # z # " -> " # whatSign(z));
